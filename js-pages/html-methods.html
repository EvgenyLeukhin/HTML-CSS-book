Мой Справочник | JavaScript | Методы работы с DOM

Поиск элементов

.querySelector()

 document.querySelector('.some-class'); Выбрать элементы по любому селектору как в CSS

.querySelectorAll()

 document.querySelector('.some-class'); Выбрать все элементы по любому селектору как в CSS

.getElementById()

 document.getElementById('some-id'); Выбрать элемент по id

.getElementsByClassName()

 document.getElementsByClassName('some-class'); Выбрать элемент по классу

.getElementsByTagName()

 document.getElementsByTagName('p'); // all elements document.getElementsByTagName('*'); // first paragraph document.getElementsByTagName('p')[0]; Выбрать элементы по названию тега
classList

.classList.add()

 someElement.classList.add('some-class'); Добавить класс к элементу someElement

.classList.remove()

 someElement.classList.remove('some-class'); Удалить класс у элемента someElement

.classList.toggle()

 someElement.classList.toggle('some-class'); Переключает класс у someElement на some-class

.classList.contains()

 someElement.classList.contains('some-class'); Проверка на наличие класса
Attributes

.setAttribute()

 someElement.setAttribute('style', 'color: red;'); Добавить атрибут или атрибут со значением

.hasAttribute()

 someElement.hasAttribute('style'); Проверка на наличие атрибута

.getAttribute()

 someElement.getAttribute('style'); Получить значение атрибута

.removeAttribute()

 someElement.removeAttribute('style'); Удалить атрибут

.className()

 someElement.className('some-class'); Добавить класс

.title()

 someElement.title('title text'); Атрибут title

.style()

 someElement.style.color = 'red';
someElement.style.backgroundColor = 'red'; 		Атрибут style. У каждого css-свойства есть свой метод в объекте style

.hidden

 someElement.hidden = true;
 Атрибут hidden. Не функция, скрывает, но не удаляет из DOM

.remove()

 someElement.remove();
 Функция, полностью удаляющаяя элемент из DOM

localStorage
Локальное хранилище 
Сохранить значение в хранилище:
localStorage.setItem('name', 'John Smith');
name - переменная (ключ), внутри которой будет храниться 'John Smith' 
Получить значение ключа в хранилище:
localStorage.getItem('name');
Если у переменной name есть сохранённое значение в localStorage, то этот метод вернёт это значение 
Удалить значение ключа в хранилище:
localStorage.removeItem('name');
Если у переменной name есть сохранённое значение в localStorage, то этот метод УДАЛИТ это значение 
Полностью очистить хранилище:
localStorage.clear();
Метод очищает все сохранённые значения в хранилище
