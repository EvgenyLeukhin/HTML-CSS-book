Math.random
Рандомный оператор, генерирующий значения от 0 до 1 в дробном значении
Например(0.6532569943) 		var a, i;
for (i=1; i<=100; i++) {
  a = Math.random(a);
  console.log(i + ' цикл: ', a);
}
Math.pow
Возведение в квадрат console.log(Math.pow(3, 5)); //243 3 возводим в 5-ую степень
Math.sqrt
Квадратный корень console.log(Math.sqrt(256)); //16
Math.abs
Модуль значения console.log(Math.abs(-123)); //123
Math.min
Минимальное значение console.log(Math.min(5, 8, 10, 2)); //2
Math.max
Максимальное значение console.log(Math.max(5, 8, 10, 2)); //10
Math.exp()
Возводит число e в указанную степень console.log(Math.exp(5)); //148.41
Math.log()
Натуральный логарифм ln числа console.log(Math.log(5)); //1.6
Math.floor
Принудительное округляет число в МЕНЬШУЮ сторону, отбрасывая дробную часть и возвращая целое число. console.log(Math.floor(123.999)); // 123
console.log(Math.floor(-123.999)); // -124
Math.ceil
Принудительное округляет число в БОЛЬШУЮ сторону, отбрасывая дробную часть и возвращая целое число. console.log(Math.ceil(123.111)); // 124
console.log(Math.ceil(-123.999)); // -123
Math.round
Округление значения console.log(Math.round(-123.527)); // - 124

.toFixed()
Количество знаков после точки var n;

.toExponential()
Перевод в экспонентальную форму (10 в степени) с кол-вом знаков после запятой в () var n;
n = 5000;
console.log(n.toExponential(2)); //5.00e+3
