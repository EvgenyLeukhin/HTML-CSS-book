React-компонент
Создание React-компонента с помощью функции

I способ создания компонента (через функцию)

// импортируем React import React from 'react'; // функция, создающая компонент const someComponent = () => {
  return(
    <h1>Hello, World!</h1>
    <h2>Hello, React!</h2>
  );
};
// экспортируем константу export default someComponent;

// импортируем React и React-Component import React, { Component } from 'react';

class App extends Component {

React-компонент со свойствами
Должен задаться аргумент функции props и через это свойство присваиваются динамические значения const SomeComponent = (props) => {
  return (
    <div className='some-class'>
      <p>My name is {props.name}</p>
      <p>My  age is {props.age}</p>
      <p>My city is {props.city}</p>
      <p>{props.children}</p>
    </div>
  );
};

Хранение данных для компонентов
Объект state для хранения данных 
class App extends Component {
  // перед рендеренгом сохраняем все данные
  компонента в объект state
  state = {
    persons: [
      { name: 'Ivan',   age: '34', city: 'Moskva'},
      { name: 'Sergei', age: '23', city: 'Tomsk' },
      { name: 'Semion', age: '19', city: 'Samara'}
    ]
  }
// а дальше вводим данные из этого объекта
  render() {
    return (
      <div className="App">
        <Person
          name={this.state.persons[0].name}
           age={this.state.persons[0].age}
          city={this.state.persons[0].city}
        />
        <Person
          name={this.state.persons[1].name}
           age={this.state.persons[1].age}
          city={this.state.persons[1].city}
        />
        <Person
          name={this.state.persons[2].name}
           age={this.state.persons[2].age}
          city={this.state.persons[2].city}
        >Some text</Person>
      </div>
    );
  }
}

class ReactComp extends React.Component {
  return React.createElement('div', this.props,
  'text ' + this.props.title);
}

Шаблон React-компонента

// Создаём функцию компонента с параметрами 
function Person(props) {
  return (
    <div className="person">
      <h1>{props.name}</h1>
      <p>Your Age: {props.age}</p>
    </div>
  );
}

// сохраняем шаблон компонента в переменную
var app = (
  <div>
     <Person name="Max" age="28" />
      <Person name="Manu" age="29" />
  </div>
);

// рендерим этот шаблон в #app
ReactDOM.render(app, document.querySelector('#app'));
