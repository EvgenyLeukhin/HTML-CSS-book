string
Строка 'Some text'
"Some text"
alert('Hello, World!');
alert(typeof('Hello, World!')); //string

Преобразование ЧИСЕЛ в СТРОКУ:

456 + '' //преобразование числа в строку
String('555'); //string
//ещё одно преобразование числа в строку
var number = 456;
number = number.toString();
console.log(typeof number); //string

Конвертация регистра

var text = 'Some Text';
//в верхний регистр
text = text.toUpperCase();
//в нижний регистр
text = text.toLowerCase();

Узнать длинну строки

console.log(text.length); //9

Вытаскивать символы из строки по индексу

console.log(text.charAt(0)); //S
console.log(text[0]); //хотя можно оперировать строкой как массивом, спец. ES5, будет такой же результат как и с .charAt(0)

Вытаскивать подстроку из строки по индексу

console.log(text.substring(5)); //text (начиная с 5-го индекса)
console.log(text.substring(5, 8)); //tex (начиная с 5-го индекса, заканчивая 8-м, не включая 8-ой)
console.log(text.slice(-2)); //xt (вытаскивает последние 2 символа из строки)
console.log(text.substr(5, 2)); //te (вытаскивает 2 символа начиная с 5-го индекса)

Вытаскивать индекс из фразы строки

console.log(text.indexOf('text')); //5 (ищет индекс первого символа в фразе)
console.log(text.lastIndexOf('text')); //8 (ищет индекс первого символа с конца фразы)

Заменить подстроку на другое значение

console.log(text.replace('Some', 'Some another')); //Some another text

Создать массив из строки

text = text.split(' '); //[Some, text]
В этом случае строка поменяет тип на объект и сохранит значение массива
console.log(typeof text); //object 		Для обозначения строкового типа данных содержимое оборачивают в кавычки (одинарные или двойные). Из нескольких строк можно "склеивать" предложения и фразы.
s Строка неизменяемый тип в отличие от number, все манипуляции над строками НЕ сохраняются, возвращаеся всегда исходная строка.
Формы записи

Экранирование кавычек или слешей (обратный слеш перед кавычками)

\ показывает интерпретатору, что кавычка или слеш после него это символ.
'Some \"text\" Some text'

Многострочная запись строки в коде

Нужно вставлять символ \ перед переходом на новую строку, при этом интерпретатор будет думать, что это одна строка, не смотря на то, что в коде это не так
'Some text1,\
Some text2,\
Some text3'

Переносы строк

Нужно вставлять n (newline) после символа \ для переноса текста на новую строку
'Some text1,\nSome text2,\nSome text3'

Табуляция

Нужно вставлять \t (tabulation) после символа \n для переноса текста на новую строку и табуляции
'Some text1,\n\t Some text2,\n\t Some text3'

Если нужны кавычки в присваевомом тексте:
alert('"Hello, World!"');
или так: alert('\"Hello, World!\"');

Преобразование СТРОК в ЧИСЛА:

var a = '5';
a = Number(a); //Конвертация строки в число

console.log(typeof(+'456')); //number (еще одна конвертация с помощью унарного оператора +)
var a = parseInt('125 px', 10);
/*Работает только с целыми числами ,10 - десятичная форма*/
console.log(typeof a); //number 125
var a = parseFloat('12.45 em', 10); //12.45 Работает c дробными числами

ПРОВЕРКА на число:

var a = '1a';
!isNaN(parseFloat(a)) && isFinite(a);
//Если а будет числом вернётся true 		Но если записать в кавычках, то будет строка alert(typeof('5'));
С числами можно производить все мат. операции.
Формы записи 16-ти:
console.log(0xabc123); //11256099
8-ми:
console.log(0123); //83
В дробных значениях, если перед точкой стоит ноль, его можно опустить:
console.log(.123); //0.123
Экспонентальная форма (х10 в степени):
console.log(1.25e4); //12500
