App.js

./src/App.js/

 // импорт react и встроенных компонентов в react import React, { Component } from 'react'; // импорт графики import logo from './logo.svg'; // импорт стилей import './App.css'; // импорт React-компонента import Person from './Person/Person';

// рендеринг компонентов и html-кода class App extends Component {
  // функция для рендеринга   render() {
    return (
      <div className="App">
          // компонент в скобках         <Person />
        <img src={logo} />
      </div>
    );
  }
}

// экспортируем класс App export default App; 		Это главный файл React-разработки.

Всю html-разметку в JSX-файлах нужно записывать с закрывающими тегами.
Все вот эти теги ( <br>, <hr>, <input>)
нужно записывать так: (<br/>, <hr/>, <input/>)
