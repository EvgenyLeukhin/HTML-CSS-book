.animate() $('.test').animate({top: '+=100px'}, 2500);

// Временные функции
$('.test').animate({top: '+=100px'}, 2500, 'swing');
$('.test').animate({top: '+=100px'}, 2500, 'linear');
 Плавно меняет какое-либо css-свойство элемента
swing - по ум*
linear - линейная

$('.test').fadeTo('slow', .33); 		Плавное появление-исчезновение-переключение прозрачности элемента, значение в милисекундах
'slow' - 600,
'normal' - 400,
'fast' - 200
