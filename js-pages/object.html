Мой Справочник | JavaScript | Объекты

Любые манипуляции со свойствами объекта изменяют сам объект. Если мы сохранили в переменную свойство объекта и поменяли его в переменной, то это свойство измениться и в самом объекте (свойства объектов находятся в одной ячейке памяти)

Объекты
Вызов объекта

Состоит из пар: имя-значение или ключ-значение

Синтаксис: {
  имя: значение,
  имя: значение,
  имя: значение
}; 		var myObject = {
  model: 'Some string',
  name: 'Some name',
  cost: 5
};
myObject.cost = 10;
//изменение значений переменных в объектах 		Объект вызывается с помощью фигурных скобок {...} - поле объекта, в которые вставляются свойства данного объекта. Свойства в объектах перечисляются через , присваиваются через :
Вызов свойств объекта
Также можно их изменять 		var myObj = {
  name: "John",
  age: 30,
  gender: "male"
};
Первый способ: console.log(myObj.name);
Второй способ: console.log(myObj['name']);

Перебор свойств объекта циклом

var obj = {
  prop1: 1,
  prop2: 2,
  prop3: 3
};
for (var i = 1; i <= 3; i++) {
  obj.['prop' + i];
}
 		Через document notation
Вложенные (многомерные) объекты var myObject = {
  model: 'Some string',
  name: {
    firstName: 'John',
    lastName: 'Smith',
    nickName: 'Robot'
  },
  cost: 5
};

Доступ к вложенному объекту:

console.log(myObject.name.firstName); . - оператор доступа
Изменение свойств из объекта myObject.name = "New name";
Удаление свойств из объекта delete myObject.age;
Запись свойств объекта в кавычках 
var demoObject = {
  'son1': 'Абырвалг',
  'son2': 'Телеграф',
  'son3': 'Телефон'
};

for (var i = 1; i <= 3 ; i++) {
  var sonName = demoObject['son' + i];
  console.log('Сын ' + i + ': ' + sonName);
}
Можно прочитать название свойств в цикле
Функции в объекте var obj = {
  p1: function() {...},
  p2: function() {...},
}
// вызов функции
obj.p1();
// обращение к функции
obj.p1; 		Если функция находится в объекте, то она называется не свойством, а методом
for in
Проверка наличия свойства в объекте "name" in myObject; //true
Название свойств var obj = {...};
for (var prop in obj) {
   prop; // сох. названия свойств
   obj[prop]; // сох. значения свойств
}
Функции в объектах
Если функция хранится в объекте, то она будет называться МЕТОДОМ var table = {
  height: 100,
  legsCount: 4,
  myFunc: function() { alert(23); }
};
table.['myFunc'];// покажет тело функции table.['myFunc']();// вызовет функцию
this
Обращение к ЭТОМУ объекту var table = {
  height: 100,
  legsCount: 4,
  myFunc: function() {
    this.height = 300;   }
}; 		this указывает на принадлежность к текущему объекту, как бы глубоко не находились его свойства
Копия объекта и копия свойств объекта
// создаём объект со свойством name
var someObj = { name: 'First-name' };

// создаём копию этого объекта в переменной copyObj
var copyObj = someObj;

// меняем name у копии
copyObj.name = 'Anoter-name';

// name у оригинала тоже меняется
console.log(someObj.name);
// но если создать копию таким способом, то перебиваться значения не будут
var copyObj2 = {
  ...someObj
} 		Так как объект и его копия занимают одну ячейку памяти, то свойства у них меняются синхронно
