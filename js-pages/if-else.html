Условные операторы if-else
Выполняет блок кода (команды) в случае ИСТИННОСТИ лог. выражения, в случае НЕИСТИННОСТИ - блок кода после else if (лог. выражение) {
    команда1;
    команда2;
    ...
} else {
    команда3;
    команда4;
    ...
} 		Лог. выражение в (...) может принимать только 2 значения: true или false

Если условий несколько, то все нужно оборачивать скобками
((...) && !(...)) {...}

ПРИМЕР:

var a, b;
a = 1;
b = 2;
if ((a + b) <= 10) { //true
    console.log('a + b <= 10');
} else {
    console.log('a + b > 10');
}
Условные операторы if-else-if
Выполняет блок кода (команды) в случае ИСТИННОСТИ лог. выражения1, в случае НЕИСТИННОСТИ - проверка лог. выражения2 и т.д. if (лог. выражение 1) {
    команда1;
    команда2;
    ...
} else if (лог. выражение 2){
    команда3;
    команда4;
    ...
} else {
    команда5;
    команда6;
    ...
} 		Может быть сколь угодно связок else if
Вложенность if в if
Если нужно проверять переменную сразу на несколько значений, то применяют вложенность условных операторов var a = prompt('a = ');
  if (a > 0) {
    if ((a >= 10) && (a < 100)) {
    console.log('a >= 10');}
    if ((a >= 100) && (a < 1000)) {
    console.log('a >= 100');}
    if (a >= 1000) {console.log('a >= 1000');}
} else if (a < 0) {console.log('a < 0');}
  else if (a = 0) {console.log('a = 0');}
  else {console.log('a неопределена');}
var a, b;
a = true;
b = true;
if (a) {
    if(b) {console.log('a и b = true')}
    if(!b) {console.log('a = true, b = false')}
} 		if(a) означает, если a = true;
if(!b) означает, если b = false;
Условные операторы switch-case
При проверке переменной на какие-либо варианты значений, удобно пользоваться командой switch case var a = 5;
switch(a) { //проверяем переменную a
    case 0: console.log('a = 0'); break;
    case 1: console.log('a = 1'); break;
    case 2: console.log('a = 2'); break;
    case 3: console.log('a = 3'); break;
    case 4: console.log('a = 4'); break;
    case 5: console.log('a = 5'); break;
    case 6: console.log('a = 6'); break;
    case 7: console.log('a = 7'); break;
    default: console.log('a > 7 или a < 0');
}; 		Использует === case - проверка на случай, их может быть сколь угодно. break - прерывание инструкции, если какой-либо case будет true, то проверка прервётся и другие случаи не будут проверяться.
default - аналог else, если не выбирется ни один case, то сработает default
Если бы не стоял break, то инструкции для a = 6, a = 7, default также бы сработали
Условный тернарный оператор
Написание условия в одну строку, единственный тернарный оператор в JS условие ? команда1 : команда2;
var a = 1;
a > 0 ? console.log('a > 0') : console.log('a <= 0');
var a = 'Some text';
a > 0 ? console.log('a > 0') :
a < 0 ? console.log('a < 0') :
a === 0 ? console.log('a = 0') :
console.log('a не определена');

Альтернативная запись

var a = 'Some text';
if (a > 0) {
  console.log('a > 0');
} else if (a < 0) {
  console.log('a < 0');
} else if (a === 0) {
  console.log('a < 0');
} else {
  console.log('a не определена');
} 		после ? выполняются команды, если условие true, после :(аналог else) выполняются команды, если условие false
