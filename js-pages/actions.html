addEventListener();
ПОИСК СОБЫТИЙ var link = document.querySelector('.user-block');
link.addEventListener('click', function(){
    console.log('Нажатие на user-block');
});
link - название переменной
addEventListener - метод
'click' - какое событие мы отслеживаем (клик мышью)
function(){} - функция, которая будет выполняться по клику

Счётчик событий <span onmouseover="counterFunc(this)"></span>
var counter = 0; //глоб. перем.
function counterFunc(element) {
    counter++;
    element.innerHTML = 'На этот текст наведено указателем мыши ' + counter + ' раз';
} 		На этот текст наведено указателем мыши 0 раз
