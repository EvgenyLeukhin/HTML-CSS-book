while
Если цикл выполнился (true) то, он всегда возвращает значения, если false, то прерывается
Условие пришется в (), так же как у оператора if 		var n = 10;
var a = 1;
while (n > a) {
  n = n - 1; // n--;
  a = a + 1; // a++;
  console.log(n - a);
}
var i = 10;
while (i--) {
  console.log(i);
} // 9,8,7,6,5,4,3,2,1,0
var i = 10;
while (--i) {
  console.log(i);
} // 9,8,7,6,5,4,3,2,1 		Пока переменная n больше переменной а, будет выполняться блок инструкций {...}
Упрощенная форма записи цикла с счётчиком внутри условия, когда счётчик i = 0, то условие не выполнится, потому что
Boolean(0) = false;
do while
Связка do-while сработает минимум 1 раз, даже если внутри while в этот первый раз будет false.

УСЛОВИЕ проверяется после ТЕЛА ЦИКЛА {...}, на практике применяется реже 		var n;
do {
  console.log('НЕ ВВОДИ 1!');
  n = parseInt(prompt('n = '));
} while (n != 1);
console.log('ЗАЧЕМ ТЫ ВВЁЛ 1'); 		Будет выполнятся блок инструкций после ключевой фразы do {...}, пока условие true
for
Цикл с упрощенной формой записи 'use strict';
for (var i = 1; i <= 10; i++) {
  var n = Math.round(10*Math.random(i));
  if (n === 10) {console.log('N = 10');}
  else {console.log(n);}
}
var i;
for (i = 1; i <= 100; i++) {
  if (i %3 == 0) {console.log(i);}
/* остаток от деления i на 3 равен нулю (кратные 3)*/
}
for (;;)
confirm('Продолжить цикл?');// бесконечный цикл 		Характеризуется 3-мя выражениями:
1 - объявление переменной с начальными значениями (инициализация);
2 - условие проверки (нельзя ставить ==, нужно <, >, >=, <=);
3 - инкремент (шаг цикла) true
Выполнение цикла - ИТЕРАЦИЯ
После завершения ЦИКЛА, значение переменной всегда будет равно, значению которое было на последней совершенной ИТЕРАЦИИ
break
Команда прерывающая цикл var i;
for (i = 1; i <= 100; i++) {
  if (i %15 == 0) {break;}
  console.log(i);
} 		Программа прервёт цикл, если условие после if выполниться, т.е. когда i странет равным 15.
continue
Делает пропуск этапа цикла var i;
for (i = 1; i <= 100; i++) {
  if (i %2 == 0) {continue;}
  console.log(i);
} 		Программа пропустит все этапы цикла при значениям i кратных двум.
Взаимодействие с массивами
Вывод элементов массива с помощью цикла var arr = [];
  arr[0] = 'A';
  arr[1] = 'B';
  arr[2] = 'C';
  arr[3] = 'D';
  arr[4] = 'E';
for (var i = 0; i < arr.length; i++) {
  console.log(arr[i]);
} 		i < arr.length - Условие, что кол-во циклов равно кол-ву элементов в массиве
Присваивание элементов массива с помощью цикла var arr = [];

} 		Можно присваивать от переменной счётчика цикла
