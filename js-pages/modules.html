export, import
Можно экспортировать-импортировать любые константы, функции, объекты, что в ES5 реализуется с помощью CommonJS const = person {
  name: 'Evgeny';
}

const = address {
  city: 'Omsk';
}

// экспортируем константы
export default person;
export address;
// импортируем константу в другом файле
import person from './person.js';
import {address} from './person.js';

// или с изменением имена
import {address as Add} from './address.js';

// импортировать всё
import *as bundled from './person.js'; 		Это новые возможности ES6.
Если export без default, то импортировать нужно в фигурных скобках и можно изменить название этой пере
function-конструктор
Конструкторы принято называть с большой буквы // создание конструктора
var Person = function(name) {
  this.name = name;
  return console.log(this.name);
};

// вызов функции-конструктора (new)
var person = new Person('Some name'); 		КОНСТРУКТОР - это функция, возвращающая объект через ключевое слово this. (Функция, создающая объект с параметрами)

Принимает свойство name и присваивает ему значение.

Можно вызвать этот конструктор через ключевое слово new.
class-конструктор
Вместо var используем ключевое слово class // создание конструктора
class Person {
  constructor(name) {
     this.name = name;
     return console.log(name);
  }
};

// вызов функции-конструктора (new)
let person = new Person('Some name'); 		В ES6 конструкторы создаются с помощью классов
